--Tworzenie tabel 

CREATE TABLE [Opiekunowie] ( --Tabela przechowująca dane rodziców/opiekunów prawnych uczniów szkoły
	[OpiekunID] [int] PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[Imię] [nvarchar](15) NOT NULL,
	[Nazwisko] [nvarchar](25) NOT NULL,
	[Telefon] [char](9) NOT NULL,
	[Miasto] [nvarchar](20) NOT NULL,
	[Kod pocztowy] [char](6) NOT NULL,
	[Ulica] [varchar](40) NOT NULL
)

CREATE TABLE [Uczniowie] ( --Tabela przechowująca dane uczniów
	[UczeńID] [int] PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[Imię] [nvarchar](15) NOT NULL,
	[Nazwisko] [nvarchar](25) NOT NULL,
	[PESEL] [char](11) NOT NULL,
	[Miasto] [nvarchar](20) NOT NULL,
	[Kod pocztowy] [char](6) NOT NULL,
	[Ulica] [nvarchar](40) NOT NULL,
	[Płeć] [char](1) NULL,
	[Data urodzenia] [date] NULL
)

CREATE TABLE [dbo].[Uczniowie opiekunów] ( --Tabela przechowująca informacje o relacji uczniów i opiekunów 
	[OpiekunID] [int] NULL,
	[UczeńID] [int] NULL
)

CREATE TABLE [dbo].[Klasy] ( --Tabela przechowująca informacje o klasach
	[KlasaID] [char](2) PRIMARY KEY NOT NULL,
	[Liczba uczniów] [tinyint] NULL,
	[Średnia ocen] [float] NULL
)

CREATE TABLE [dbo].[Charakterystyka klasy] (--Tabela przechowująca informacje o relacji uczniów i klas 
	[KlasaID] [char](2) NOT NULL,
	[UczeńID] [int] NOT NULL
)

CREATE TABLE [dbo].[Przedmioty] (--Przedmioty i ich nazwa
	[PrzedmiotID] [int] PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[Nazwa] [nvarchar](20) NULL
)

CREATE TABLE [dbo].[Pracownicy] (--Pracownicy i ich dane 
	[PracownikID] [int] PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[Imię] [nvarchar](15) NOT NULL,
	[Nazwisko] [nvarchar](25) NOT NULL,
	[Telefon] [char](9) NOT NULL,
	[Numer dowodu] [char](9) NOT NULL,
	[Miasto] [nvarchar](20) NOT NULL,
	[Kod pocztowy] [char](6) NOT NULL,
	[Ulica] [varchar](40) NOT NULL
)

CREATE TABLE [dbo].[Specjalizacje] (--Specjalizacje i ich nazwy
	[SpecjalizacjaID] [int] PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[Nazwa] [varchar](20) NULL
)

CREATE TABLE [dbo].[Specjalizacje nauczycieli] ( --Relacja pracowników i ich specjalizacji
	[PracownikID] [int] NULL,
	[SpecjalizacjaID] [int] NULL
)

CREATE TABLE [dbo].[Zajęcia] (--Tabela trzymająca informacje o tym jaka klasa, z jakiego przedmiotu i który pracownik prwadzi zajęcia 
	[ZajęcieID] [int] PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[KlasaID] [char](2) NOT NULL,
	[PrzedmiotID] [int] NOT NULL,
	[PracownikID] [int] NOT NULL,
	[Sala] [tinyint] NOT NULL

CREATE TABLE [dbo].[Oceny] (--Oceny uczniów, z możliwością odwolania się do poprzedniej (oceny przed poprawy)

	[UczeńID] [int] NOT NULL,
	[ZajęcieID] [int] NOT NULL,
	[Ocena] [float] NOT NULL,

	[Poprzednia ocena] [int] NULL

CREATE TRIGGER [dbo].[MinusUczen] ON [dbo].[Charakterystyka klasy] --Zmiejszenie ilości uczniów o 1 w tabeli Klasa
AFTER DELETE
AS 
DECLARE @IdKlasy char(2)
DECLARE @LiczbaUczniow int;
SET @IdKlasy = (Select deleted.KlasaID from deleted)
SET @LiczbaUczniow = (SELECT Klasy.[Liczba uczniów] FROM KLASY WHERE KlasaID=@IdKlasy)
IF (@LiczbaUczniow > 0)
	BEGIN
		UPDATE Klasy SET Klasy.[Liczba uczniów]-=1 WHERE Klasy.KlasaID=@IdKlasy
	END
--------------

--------------
CREATE TRIGGER [dbo].[PlusUczen] ON [dbo].[Charakterystyka klasy] --Zwiększenie ilości uczniów o 1 w tabeli Klasa
AFTER INSERT
AS
	BEGIN
		UPDATE Klasy SET Klasy.[Liczba uczniów]+=1 WHERE Klasy.KlasaID IN (Select inserted.KlasaID from inserted)
	END
--------------

--------------
CREATE TRIGGER [dbo].[SredniaOcen] ON [dbo].[Oceny] --Aktualizacja średniej w tabeli Klasy
AFTER INSERT
AS
Declare @Klasa char(2)
Set @Klasa = (select Zajęcia.KlasaID from inserted Join Zajęcia ON Zajęcia.ZajęcieID=inserted.ZajęcieID)
Declare @Srednia FLOAT
Set @Srednia = (SELECT Round(Avg(Oceny.Ocena),2) FROM Oceny JOIN Zajęcia 
ON Oceny.ZajęcieID=Zajęcia.ZajęcieID where zajęcia.KlasaID = @Klasa)
Update Klasy set Klasy.[Średnia ocen]=@Srednia where KlasaID=@Klasa
--------------

CREATE TRIGGER [dbo].[PoprawnyPracownik] ON [dbo].[Zajęcia] --Sprawdzenie czy wprowadzany pracownik do zajęć posiada odpowiednią specjalizację
AFTER INSERT
AS
	Declare @IdNauczyciel int
	Declare @NazwaPrzedmiot varchar(20)
	Declare @IdSpecjalizacja int
	Declare @Flaga int; 
	Set @NazwaPrzedmiot = (Select Przedmioty.Nazwa from Przedmioty where Przedmioty.PrzedmiotID=(select PrzedmiotID from inserted))
	Set @IdNauczyciel = (select inserted.PracownikID from inserted)
	Set @IdSpecjalizacja = (Select Specjalizacje.SpecjalizacjaID from Specjalizacje where Nazwa=@NazwaPrzedmiot)
	set @Flaga = (SELECT 
	CASE WHEN EXISTS 
		(SELECT  SN.SpecjalizacjaID
		 FROM [Specjalizacje nauczycieli] as SN 
		 WHERE SN.SpecjalizacjaID=@IdSpecjalizacja and SN.PracownikID=@IdNauczyciel)
		THEN 1 
		ELSE 0 
	END )
	IF(@Flaga = 0) 
		BEGIN
		ROLLBACK
			RAISERROR('Pracownik nie specjalizuje się w tym przedmiocie!',16,1)
		END

CREATE FUNCTION [dbo].[FUNC_TlumaczenieOcen] (@TempOcena FLOAT)
RETURNS varchar(200)
AS 
BEGIN 
	IF @TempOcena = 1 
		BEGIN 
			RETURN 'ndst'
		END
	ELSE IF @TempOcena = 1.5
		BEGIN 
			RETURN '+ndst'
		END
	ELSE IF @TempOcena = 1.75
		BEGIN 
			RETURN '-dop'
		END
	ELSE IF @TempOcena = 2
		BEGIN 
			RETURN 'dop'
		END
	ELSE IF @TempOcena = 2.5
		BEGIN 
			RETURN '+dop'
		END
	ELSE IF @TempOcena = 2.75
		BEGIN 
			RETURN '-dst'
		END
	ELSE IF @TempOcena = 3
		BEGIN 
			RETURN 'dst'
		END
	ELSE IF @TempOcena = 3.5
		BEGIN 
			RETURN '+dst'
		END
	ELSE IF @TempOcena = 3.75
		BEGIN 
			RETURN '-db'
		END
	ELSE IF @TempOcena = 4
		BEGIN 
			RETURN 'db'
		END
	ELSE IF @TempOcena = 4.5
		BEGIN 
			RETURN '+db'
		END
	ELSE IF @TempOcena = 4.75
		BEGIN 
			RETURN '-bdb'
		END
	ELSE IF @TempOcena = 5
		BEGIN 
			RETURN 'bdb'
		END
	ELSE IF @TempOcena = 5.5
		BEGIN 
			RETURN '+bdb'
		END
	ELSE IF @TempOcena = 5.75
		BEGIN 
			RETURN '-cel'
		END
	ELSE IF @TempOcena = 6
		BEGIN 
			RETURN 'cel'
		END
	RETURN 'error'
END

CREATE FUNCTION [dbo].[Lista uczniów]  (@IdKlasy Char(2))
RETURNS TABLE
AS
RETURN 
(
   Select ROW_NUMBER() OVER(ORDER BY U.Nazwisko) AS Numer, U.Imię, U.Nazwisko From Uczniowie AS U 
   JOIN [Charakterystyka klasy] AS CK ON CK.UczeńID=U.UczeńID
   Where CK.KlasaID = @IdKlasy
)
GO
--------------

--------------
CREATE FUNCTION [dbo].[NajlepszyUczeńKlasy]  (@IdKlasy Char(2))
RETURNS TABLE
AS
RETURN 
(
   WITH CTE (NajlepszaSrednia) 
AS 
(
	Select MAX(SO.SredniaOcena) FROM 
		(
			Select CK.UczeńID, Avg(O.Ocena) as SredniaOcena 
			From Oceny AS O 
			JOIN Uczniowie AS U ON O.UczeńID=U.UczeńID
			JOIN [Charakterystyka klasy] AS CK ON CK.UczeńID=U.UczeńID
			WHERE CK.KlasaID = @IdKlasy
			GROUP BY CK.UczeńID
		) AS SO
)
Select U.Imię, U.Nazwisko, Avg(O.Ocena) AS Srednia FROM Klasy as K 
JOIN [Charakterystyka klasy] AS CK ON CK.KlasaID=K.KlasaID
JOIN Uczniowie AS U ON U.UczeńID=CK.UczeńID
JOIN Oceny AS O ON O.UczeńID=U.UczeńID
WHERE K.KlasaID = @IdKlasy
GROUP BY U.Imię, U.Nazwisko
HAVING Avg(O.Ocena) = (SELECT NajlepszaSrednia FROM CTE)
)
GO

--------------

--------------
CREATE FUNCTION [dbo].[Specjalizacja]  (@NazwaSpec VARCHAR(20))
RETURNS TABLE
AS
RETURN 
(
   SELECT P.Imię, P.Nazwisko FROM Pracownicy AS P
   JOIN [Specjalizacje nauczycieli] AS SN ON SN.PracownikID=P.PracownikID
   JOIN Specjalizacje AS S ON S.SpecjalizacjaID=SN.SpecjalizacjaID
   WHERE S.Nazwa = @NazwaSpec
)
GO

CREATE PROC [dbo].[PROC_ZestawienieDlaOpiekuna](@IdOpiekuna INT) --wyświetla oceny uczniów opiekunów z "przetłumaczonymi" ocenami 
AS
SELECT U.Imię, U.Nazwisko, dbo.FUNC_TlumaczenieOcen(O.Ocena) As Ocena, P.Nazwa 
FROM Uczniowie AS U 
JOIN [Uczniowie opiekunów] AS UO ON UO.UczeńID=U.UczeńID
JOIN Oceny AS O ON O.UczeńID=U.UczeńID
JOIN Zajęcia AS Z ON O.ZajęcieID=Z.ZajęcieID
JOIN Przedmioty AS P ON P.PrzedmiotID=Z.PrzedmiotID
WHERE UO.OpiekunID = @IdOpiekuna
--------------

CREATE PROC [dbo].[PROC_NajlepszyUczenKlasy] (@IdKlasy char(2)) --wskazuje ucznia/ów z najlepszą średnią
AS
;WITH CTE (NajlepszaSrednia) 
AS 
(
	Select MAX(SO.SredniaOcena) FROM 
		(
			Select CK.UczeńID, Avg(O.Ocena) as SredniaOcena 
			From Oceny AS O 
			JOIN Uczniowie AS U ON O.UczeńID=U.UczeńID
			JOIN [Charakterystyka klasy] AS CK ON CK.UczeńID=U.UczeńID
			WHERE CK.KlasaID = @IdKlasy
			GROUP BY CK.UczeńID
		) AS SO
)
Select U.Imię, U.Nazwisko, Avg(O.Ocena) AS Srednia FROM Klasy as K 
JOIN [Charakterystyka klasy] AS CK ON CK.KlasaID=K.KlasaID
JOIN Uczniowie AS U ON U.UczeńID=CK.UczeńID
JOIN Oceny AS O ON O.UczeńID=U.UczeńID
WHERE K.KlasaID = @IdKlasy
GROUP BY U.Imię, U.Nazwisko
HAVING Avg(O.Ocena) = (SELECT NajlepszaSrednia FROM CTE)

CREATE PROC [dbo].[PROC_SredniaOcen] (@IdUcznia INT, @IdZajec INT) --Wyświetla wskazanego ucznia średnią ocen z danych zajęć
AS 
Select U.Imię, U.Nazwisko, ROUND(Avg(O.Ocena), 2) FROM Uczniowie AS U JOIN Oceny AS O ON 
O.UczeńID=U.UczeńID Where U.UczeńID = @IdUcznia AND O.ZajęcieID = @IdZajec GROUP BY U.Imię, U.Nazwisko
